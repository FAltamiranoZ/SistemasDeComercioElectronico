package pojoproducto;

import java.math.BigDecimal;
import javax.xml.datatype.XMLGregorianCalendar;

/**
 *
 * @author rggh
 */
public class PojoProducto {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) 
    {
        boolean primero = true;
        XMLGregorianCalendar fecha = null;
        wsclte.Category cat = new wsclte.Category();
        wsclte.Product prod = new wsclte.Product();
         
         java.util.List<wsclte.Product> lista = findAll();
         for( wsclte.Product p:lista)
         {
             System.out.println(p.getId() + " " + p.getName() + " " + p.getDescription() + " " + p.getPrice() + " Cat:" + p.getCategoryId().getId() + " " + p.getCategoryId().getName() );
             if(primero)
             {
                cat=p.getCategoryId();
                fecha = p.getLastUpdate();
                primero = false;
             }
         }
         
       prod.setCategoryId(cat);
       prod.setName("Producto nuevo");
       prod.setDescription("Descrip. producto nuevo");
       prod.setPrice(new BigDecimal(25.0));
       prod.setLastUpdate(fecha);
      
       create(prod);
         
    }

    private static java.util.List<wsclte.Product> findAll() {
        wsclte.NewWebService_Service service = new wsclte.NewWebService_Service();
        wsclte.NewWebService port = service.getNewWebServicePort();
        return port.findAll();
    }

    private static void create(wsclte.Product entity) {
        wsclte.NewWebService_Service service = new wsclte.NewWebService_Service();
        wsclte.NewWebService port = service.getNewWebServicePort();
        port.create(entity);
    }
